cd- mudar diretório
dir- listar
mkdir- criar diretório
del/rmdir- deletar diretório
sha1- encriptação, gera um comjunto de caracteres identificador de 40 dígitos
ls- listar
ctrl l- limpar
git init- inicializar reposório
ls -a- mostar pasta oculta
git status- status do arquivo
mv- mover
git add "nome do arquivo"- adiciona só o arquivo citado
git add *- adiciona todas as modificações
git commit -m- comitar "escrever instrução"
git push origin - empurrar um repositório no Git
git pull origin - puxar um repositório do Git
git clone- clonar um repositório
